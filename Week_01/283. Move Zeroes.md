public void moveZeroes(int[] nums) {
    int slowPoint = 0;
    for (int quickPoint = 0; quickPoint < nums.length; quickPoint++) {
        if (nums[quickPoint] != 0) {
            nums[slowPoint] = nums[quickPoint];
            slowPoint++;
        }
    }
    for (int j = slowPoint; j < nums.length; j++) {
        nums[j] = 0;
    }
}
